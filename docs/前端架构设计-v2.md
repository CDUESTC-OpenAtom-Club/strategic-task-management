```

│                         Browser                             │
└───────────────────────────┬─────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐

                            │
        ┌───────────────────┼───────────────────┐
        ▼                   ▼                   ▼
┌───────────────┐   ┌───────────────┐   ┌───────────────┐
│   Pages       │   │   Widgets     │   │  Processes    │
│   (路由页面)   │   │   (页面组件)   │   │  (流程编排)   │
└───────────────┘   └───────────────┘   └───────────────┘
        │                   │                   │
        └───────────────────┼───────────────────┘
                            ▼
                  ┌───────────────┐
                  │   Features    │
                  │  (业务特性)    │
                  └───────────────┘
                            │
                            ▼
                  ┌───────────────┐
                  │   Entities    │
                  │  (领域模型)    │
                  └───────────────┘
                            │
                            ▼
                  ┌───────────────┐
                  │    Shared     │
                  │  (共享资源)    │
                  └───────────────┘
```

### FSD 分层说明

| 层级          | 职责             | 可依赖                                         | 示例                       |
| ------------- | ---------------- | ---------------------------------------------- | -------------------------- |
| **app**       | 应用配置、根组件 | 所有层                                         | App.vue, router, providers |
| **processes** | 跨特性流程编排   | features, entities, shared                     | 指标下发流程               |
| **pages**     | 路由级页面组件   | processes, widgets, features, entities, shared | IndicatorListPage          |
| **widgets**   | 页面级可组合组件 | features, entities, shared                     | IndicatorTable             |
| **features**  | 自包含业务特性   | entities, shared                               | strategic-indicator        |
| **entities**  | 领域模型共享     | shared                                         | User, Organization         |
| **shared**    | 可复用资源和工具 | 无                                             | UI组件, 工具函数           |

## 设计原则

### 1. 特性切片设计 (FSD)

**核心理念**：按业务特性组织代码，而非按技术层次

**优势**：

- ✅ 特性自包含，易于维护
- ✅ 清晰的依赖关系
- ✅ 避免循环依赖
- ✅ 便于团队协作

### 2. 单向依赖规则

```
app → pages → widgets → features → entities → shared
```

**依赖示例**：

```typescript
// ✅ 正确：Page 依赖 Feature
import { IndicatorList } from '@/features/strategic-indicator'

// ✅ 正确：Feature 依赖 Entity
import { Indicator } from '@/entities/indicator'

// ✅ 正确：Feature 依赖 Shared
import { DataTable } from '@/shared/ui'

// ❌ 错误：Entity 依赖 Feature
// import { something } from '@/features/xxx';

// ❌ 错误：Shared 依赖任何层
// import { something } from '@/entities/xxx';
```

### 3. 特性自包含

每个 Feature 包含完整的业务逻辑：

```
feature-name/
├── model/        # 业务逻辑和状态
├── lib/          # 内部工具（私有）
├── api/          # API 调用
├── ui/           # UI 组件（可导出）
└── index.ts      # 公共 API
```

### 4. 类型安全

- 使用 TypeScript 定义所有类型
- 使用 Zod 进行运行时验证
- 避免使用 `any` 类型

```typescript
// processes/indicator-distribution/lib/distribution-process.ts

import { indicatorApi } from '@/entities/indicator'
import { notificationApi } from '@/entities/notification'
import { orgApi } from '@/entities/organization'
import { validateIndicators } from './validation'
import { sendDistributionNotifications } from './notification'
import type { DistributionRequest } from '../model/types'

export async function distributeIndicators(request: DistributionRequest): Promise<void> {
  // 1. 验证指标可以下发
  await validateIndicators(request.indicatorIds)

  // 2. 验证目标组织
  await validateOrganizations(request.targetOrgIds)

  // 3. 执行下发
  const results = await indicatorApi.distribute({
    indicatorIds: request.indicatorIds,
    targetOrgIds: request.targetOrgIds
  })

  // 4. 发送通知
  await sendDistributionNotifications(results)

  // 5. 记录审计日志
  await auditLog({
    action: 'INDICATOR_DISTRIBUTED',
    details: {
      indicatorIds: request.indicatorIds,
      targetOrgIds: request.targetOrgIds,
      results
    }
  })

  return results
}

// 异常处理
export async function rollbackDistribution(distributionId: string): Promise<void> {
  await indicatorApi.withdrawDistribution(distributionId)
  await notificationApi.sendRollbackNotification(distributionId)
}
```

```typescript
// features/strategic-indicator/model/store.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { indicatorApi } from './api'
import type { Indicator, IndicatorFilters } from './types'

export const useIndicatorStore = defineStore('indicator', () => {
  // State
  const indicators = ref<Indicator[]>([])
  const currentIndicator = ref<Indicator | null>(null)
  const filters = ref<IndicatorFilters>({})
  const loading = ref(false)
  const error = ref<string | null>(null)

  // Getters
  const pendingIndicators = computed(() =>
    indicators.value.filter(i => i.status === 'PENDING_APPROVAL')
  )

  const distributedIndicators = computed(() =>
    indicators.value.filter(i => i.status === 'DISTRIBUTED')
  )

  const indicatorsByPlan = computed(
    () => (planId: string) => indicators.value.filter(i => i.planId === planId)
  )

  // Actions
  async function fetchIndicators(params?: IndicatorFilters) {
    loading.value = true
    error.value = null
    try {
      const data = await indicatorApi.query({
        ...filters.value,
        ...params
      })
      indicators.value = data.items
    } catch (e) {
      error.value = e instanceof Error ? e.message : '获取指标列表失败'
      throw e
    } finally {
      loading.value = false
    }
  }

  async function fetchIndicator(id: string) {
    loading.value = true
    error.value = null
    try {
      const indicator = await indicatorApi.getById(id)
      currentIndicator.value = indicator
      return indicator
    } catch (e) {
      error.value = e instanceof Error ? e.message : '获取指标详情失败'
      throw e
    } finally {
      loading.value = false
    }
  }

  async function createIndicator(payload: CreateIndicatorDTO) {
    loading.value = true
    error.value = null
    try {
      const indicator = await indicatorApi.create(payload)
      indicators.value.push(indicator)
      return indicator
    } catch (e) {
      error.value = e instanceof Error ? e.message : '创建指标失败'
      throw e
    } finally {
      loading.value = false
    }
  }

  async function updateIndicator(id: string, payload: UpdateIndicatorDTO) {
    loading.value = true
    error.value = null
    try {
      const indicator = await indicatorApi.update(id, payload)
      const index = indicators.value.findIndex(i => i.id === id)
      if (index !== -1) {
        indicators.value[index] = indicator
      }
      if (currentIndicator.value?.id === id) {
        currentIndicator.value = indicator
      }
      return indicator
    } catch (e) {
      error.value = e instanceof Error ? e.message : '更新指标失败'
      throw e
    } finally {
      loading.value = false
    }
  }

  async function deleteIndicator(id: string) {
    loading.value = true
    error.value = null
    try {
      await indicatorApi.delete(id)
      indicators.value = indicators.value.filter(i => i.id !== id)
      if (currentIndicator.value?.id === id) {
        currentIndicator.value = null
      }
    } catch (e) {
      error.value = e instanceof Error ? e.message : '删除指标失败'
      throw e
    } finally {
      loading.value = false
    }
  }

  function setFilters(newFilters: Partial<IndicatorFilters>) {
    filters.value = { ...filters.value, ...newFilters }
  }

  function resetFilters() {
    filters.value = {}
  }

  return {
    // State
    indicators,
    currentIndicator,
    filters,
    loading,
    error,

    // Getters
    pendingIndicators,
    distributedIndicators,
    indicatorsByPlan,

    // Actions
    fetchIndicators,
    fetchIndicator,
    createIndicator,
    updateIndicator,
    deleteIndicator,
    setFilters,
    resetFilters
  }
})
```

```typescript
export function formatDate(date: string | Date, format = 'YYYY-MM-DD'): string {
  const d = typeof date === 'string' ? new Date(date) : date
  // 使用 dayjs 或其他库格式化
  return dayjs(d).format(format)

}

export function formatRelativeTime(date: string | Date): string {
  const d = typeof date === 'string' ? new Date(date) : date
  const now = new Date()
  const diff = now.getTime() - d.getTime()

  const minutes = Math.floor(diff / 60000)
  const hours = Math.floor(minutes / 60)
  const days = Math.floor(hours / 24)

  if (days > 0) return `${days}天前`
  if (hours > 0) return `${hours}小时前`
  if (minutes > 0) return `${minutes}分钟前`
  return '刚刚'
}
```

```typescript
// shared/lib/storage/pinia-persist.ts

import { watch } from 'vue'
import type { Store } from 'pinia'

export function persistStore<T extends Store>(store: T, key: string, paths?: string[]) {
  // 从 localStorage 恢复
  const saved = localStorage.getItem(key)
  if (saved) {
    try {
      const data = JSON.parse(saved)
      store.$patch(data)
    } catch (e) {
      console.error('Failed to restore store:', e)
    }
  }

  // 监听变化并保存
  store.$subscribe((mutation, state) => {
    const dataToSave = paths
      ? paths.reduce((acc, path) => {
          acc[path] = state[path]
          return acc
        }, {} as any)
      : state

    localStorage.setItem(key, JSON.stringify(dataToSave))
  })
}

// 使用
import { useAuthStore } from '@/features/auth'
```

### API Client

```typescript
// shared/lib/api/client.ts

import axios, { AxiosError, AxiosInstance, AxiosRequestConfig } from 'axios'
import { handleError } from './errorHandler'
import { retryRequest } from './retry'

export function createApiClient(config?: AxiosRequestConfig): AxiosInstance {
  const client = axios.create({
    baseURL: import.meta.env.VITE_API_BASE_URL,
    timeout: 30000,
    ...config
  })

  // 请求拦截器
  client.interceptors.request.use(
    config => {
      const token = localStorage.getItem('token')
      if (token) {
        config.headers.Authorization = `Bearer ${token}`
      }
      return config
    },
    error => Promise.reject(error)
  )

  // 响应拦截器
  client.interceptors.response.use(
    response => response.data,
    async (error: AxiosError) => {
      // 尝试重试
      if (await retryRequest(error)) {
        return client.request(error.config)
      }

      // 处理错误
      handleError(error)
      return Promise.reject(error)
    }
  )

  return client
}

export const apiClient = createApiClient()
```

### 错误处理

```typescript
// shared/lib/api/errorHandler.ts

import { ElMessage } from 'element-plus'
import type { AxiosError } from 'axios'

export function handleError(error: AxiosError) {
  if (error.response) {
    const { status, data } = error.response

    switch (status) {
      case 400:
        ElMessage.error(data.message || '请求参数错误')
        break
      case 401:
        ElMessage.error('未登录，请先登录')
        // 跳转到登录页
        window.location.href = '/login'
        break
      case 403:
        ElMessage.error('无权限访问')
        break
      case 404:
        ElMessage.error('请求的资源不存在')
        break
      case 500:
        ElMessage.error('服务器错误，请稍后重试')
        break
      default:
        ElMessage.error(data.message || '请求失败')
    }
  } else if (error.request) {
    ElMessage.error('网络错误，请检查网络连接')
  } else {
    ElMessage.error('请求配置错误')
  }
}
```

### 重试机制

```typescript
// shared/lib/api/retry.ts

import type { AxiosError } from 'axios'

const MAX_RETRY = 3
const RETRY_DELAY = 1000

export async function retryRequest(error: AxiosError): Promise<boolean> {
  const config = error.config as any & { _retry?: number; _retryCount?: number }

  // 只重试网络错误和 5xx 错误
  if (!error.response || error.response.status < 500) {
    return false
  }

  // 初始化重试次数
  if (!config._retry) {
    config._retry = true
    config._retryCount = 0
  }

  // 检查重试次数
  if (config._retryCount >= MAX_RETRY) {
    return false
  }

  // 增加重试次数
  config._retryCount++

  // 延迟重试
  await new Promise(resolve => setTimeout(resolve, RETRY_DELAY * config._retryCount))

  return true
}
```

```typescript
// app/router/guards.ts

import type { Router } from 'vue-router'
import { useAuthStore } from '@/features/auth'

export function setupRouterGuards(router: Router) {
  // 认证守卫
  router.beforeEach((to, from, next) => {
    const authStore = useAuthStore()
    const requiresAuth = to.matched.some(record => record.meta.requiresAuth !== false)

    if (requiresAuth && !authStore.isLoggedIn) {
      next({
        name: 'Login',
        query: { redirect: to.fullPath }
      })
      return
    }

    next()
  })

  // 权限守卫
  router.beforeEach((to, from, next) => {
    const authStore = useAuthStore()
    const permission = to.meta.permission as string | undefined

    if (permission && !authStore.hasPermission(permission)) {
      next({ name: 'Forbidden' })
      return
    }

    next()
  })

  // 页面标题
  router.afterEach(to => {
    const title = to.meta.title as string | undefined
    if (title) {
      document.title = `${title} - 战略任务管理系统`
    }
  })
}
```

```typescript
export function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timeoutId: ReturnType<typeof setTimeout>

  return function (...args: Parameters<T>) {
    clearTimeout(timeoutId)
    timeoutId = setTimeout(() => fn(...args), delay)
  }
}
```
