```typescript
// shared/lib/api/client.ts
import axios, { AxiosInstance, AxiosRequestConfig } from 'axios'
import { handleError } from './error-handler'
import { withRetry } from './retry'

export function createApiClient(config?: AxiosRequestConfig): AxiosInstance {
  const client = axios.create({
    baseURL: import.meta.env.VITE_API_BASE_URL,
    timeout: 30000,
    ...config
  })

  // 请求拦截器
  client.interceptors.request.use(
    config => {
      const token = localStorage.getItem('token')
      if (token) {
        config.headers.Authorization = `Bearer ${token}`
      }
      return config
    },
    error => Promise.reject(error)
  )

  // 响应拦截器
  client.interceptors.response.use(
    response => response.data,
    async error => {
      // 尝试重试
      if (await withRetry(error)) {
        return client.request(error.config)
      }
      // 处理错误
      handleError(error)
      return Promise.reject(error)
    }
  )

  return client
}

export const apiClient = createApiClient()
```

```typescript
// shared/lib/api/retry.ts
import type { AxiosError } from 'axios'

interface RetryConfig {
  maxRetries?: number
  retryDelay?: number
  retryableStatuses?: number[]
}

const DEFAULT_RETRY_CONFIG: Required<RetryConfig> = {
  maxRetries: 3,
  retryDelay: 1000,
  retryableStatuses: [408, 429, 500, 502, 503, 504]
}

export async function withRetry(error: AxiosError, config?: RetryConfig): Promise<boolean> {
  const cfg = { ...DEFAULT_RETRY_CONFIG, ...config }
  const axiosConfig = error.config as any & {
    _retry?: boolean
    _retryCount?: number
  }

  // 检查是否应该重试
  if (!error.response) {
    // 网络错误，可以重试
  } else if (!cfg.retryableStatuses.includes(error.response.status)) {
    return false
  }

  // 初始化重试计数
  if (!axiosConfig._retry) {
    axiosConfig._retry = true
    axiosConfig._retryCount = 0
  }

  // 检查重试次数
  if (axiosConfig._retryCount >= cfg.maxRetries) {
    return false
  }

  // 增加重试次数
  axiosConfig._retryCount++

  // 延迟重试
  await new Promise(resolve => setTimeout(resolve, cfg.retryDelay * axiosConfig._retryCount!))

  return true
}
```

```typescript
// shared/lib/api/error-handler.ts
import { ElMessage } from 'element-plus'
import type { AxiosError } from 'axios'

export function handleError(error: AxiosError): void {
  if (error.response) {
    const { status, data } = error.response

    const messageMap: Record<number, string> = {
      400: '请求参数错误',
      401: '未登录，请先登录',
      403: '无权限访问',
      404: '请求的资源不存在',
      500: '服务器错误，请稍后重试'
    }

    const message = (data as any)?.message || messageMap[status] || '请求失败'
    ElMessage.error(message)

    // 401特殊处理
    if (status === 401) {
      localStorage.removeItem('token')
      window.location.href = '/login'
    }
  } else if (error.request) {
    ElMessage.error('网络错误，请检查网络连接')
  } else {
    ElMessage.error('请求配置错误')
  }
}
```

```typescript
export function formatDate(date: string | Date, format = 'YYYY-MM-DD'): string {
  const d = typeof date === 'string' ? new Date(date) : date
  // 使用dayjs或原生实现
  const year = d.getFullYear()
  const month = String(d.getMonth() + 1).padStart(2, '0')
  const day = String(d.getDate()).padStart(2, '0')

  if (format === 'YYYY-MM-DD') {
    return `${year}-${month}-${day}`
  }
  // 其他格式...
}

export function formatDateTime(date: string | Date): string {
  return formatDate(date, 'YYYY-MM-DD HH:mm:ss')
}

export function formatRelativeTime(date: string | Date): string {
  const d = typeof date === 'string' ? new Date(date) : date
  const now = new Date()
  const diff = now.getTime() - d.getTime()

  const minutes = Math.floor(diff / 60000)
  const hours = Math.floor(minutes / 60)
  const days = Math.floor(hours / 24)

  if (days > 0) return `${days}天前`
  if (hours > 0) return `${hours}小时前`
  if (minutes > 0) return `${minutes}分钟前`
  return '刚刚'
}
```

### Task 3.4: 创建通用Store模板

```typescript
// shared/lib/store/createCrudStore.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export interface CrudStoreOptions<T, Filters, CreateDTO, UpdateDTO> {
  name: string
  queryList: (filters?: Filters) => Promise<PaginatedResult<T>>
  getById: (id: string) => Promise<T>
  create: (data: CreateDTO) => Promise<T>
  update: (id: string, data: UpdateDTO) => Promise<T>
  delete: (id: string) => Promise<void>
}

export function createCrudStore<T, Filters, CreateDTO, UpdateDTO>(
  options: CrudStoreOptions<T, Filters, CreateDTO, UpdateDTO>
) {
  return defineStore(options.name, () => {
    const items = ref<T[]>([])
    const current = ref<T | null>(null)
    const filters = ref<Filters>({} as Filters)
    const loading = ref(false)
    const error = ref<string | null>(null)

    const itemsById = computed(() =>
      Object.fromEntries(items.value.map(item => [(item as any).id, item]))
    )

    async function fetchList(params?: Filters) {
      loading.value = true
      error.value = null
      try {
        const data = await options.queryList({ ...filters.value, ...params })
        items.value = data.items
      } catch (e) {
        error.value = e instanceof Error ? e.message : '获取失败'
        throw e
      } finally {
        loading.value = false
      }
    }

    async function fetchById(id: string) {
      loading.value = true
      error.value = null
      try {
        current.value = await options.getById(id)
        return current.value
      } catch (e) {
        error.value = e instanceof Error ? e.message : '获取失败'
        throw e
      } finally {
        loading.value = false
      }
    }

    async function create(data: CreateDTO) {
      loading.value = true
      error.value = null
      try {
        const item = await options.create(data)
        items.value.push(item)
        return item
      } catch (e) {
        error.value = e instanceof Error ? e.message : '创建失败'
        throw e
      } finally {
        loading.value = false
      }
    }

    async function update(id: string, data: UpdateDTO) {
      loading.value = true
      error.value = null
      try {
        const item = await options.update(id, data)
        const index = items.value.findIndex(i => (i as any).id === id)
        if (index !== -1) {
          items.value[index] = item
        }
        if (current.value && (current.value as any).id === id) {
          current.value = item
        }
        return item
      } catch (e) {
        error.value = e instanceof Error ? e.message : '更新失败'
        throw e
      } finally {
        loading.value = false
      }
    }

    async function remove(id: string) {
      loading.value = true
      error.value = null
      try {
        await options.delete(id)
        items.value = items.value.filter(i => (i as any).id !== id)
        if (current.value && (current.value as any).id === id) {
          current.value = null
        }
      } catch (e) {
        error.value = e instanceof Error ? e.message : '删除失败'
        throw e
      } finally {
        loading.value = false
      }
    }

    function setFilters(newFilters: Partial<Filters>) {
      filters.value = { ...filters.value, ...newFilters }
    }

    function resetFilters() {
      filters.value = {} as Filters
    }

    return {
      items,
      current,
      filters,
      loading,
      error,
      itemsById,
      fetchList,
      fetchById,
      create,
      update,
      remove,
      setFilters,
      resetFilters
    }
  })
}

// 使用示例

import * as indicatorApi from '../api'

  name: 'indicator',
  queryList: indicatorApi.queryList,
  getById: indicatorApi.getById,
  create: indicatorApi.create,
  update: indicatorApi.update,
  delete: indicatorApi.remove
})
```
